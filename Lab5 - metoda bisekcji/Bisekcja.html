def metodaBisekcji(y,x1,x2):
    red = [[x1,y(x1)]] #x2
    green = [[x2,y(x2)]] #x1
    yellow =[] #root
#-----------------------------funckja--------------------------------   
    if y(x1)*y(x2) > 0:
        print("brak pierwiastka w danej sekcji")
        return 0
        
    if y(x1) == 0:
        print("x1 jednym pierwiastkw")
        return 0
        
    if y(x2) == 0:
        print("x2 jednym pierwiastkw")
        return 0
    
    #jeeli adne z powyszych ifw si nie speni
    #znacz to e e pierwiastek znajduje si w podanym przedziale
    for i in range(iloscIteracji):
        c = (x1+x2)/2 #bisekcja
        if y(c)*y(x1) < 0: #pierwiastek znajduje si pomidzy c a x1 wic x2 przyjumje warto c
            x2 = c
            green.append([x2,y(x2)])
        else:              #pierwiastek znajduje si pomidzy c a x2 wic x1 przyjumje warto c
            x1 = c
            red.append([x1,y(x1)])
        if abs(y(x1)) < tol:
            print("ilo wykonanych iteracji: ",i) 
            break
#------------------------------------------------------------- 
    print("pierwiastek znajduje si w x: ",x1)
    yellow.append([x1,y(x1)]) #root
    x_list = np.linspace(0,2.5,100)
    x_list[0] = 0.00001
    fplot = list(map(y, x_list))
    plt.plot(x_list, fplot)
    #x1
    x, y = list(zip(*green))
    plt.scatter(x, y, c='r')
    #x2
    x, y = list(zip(*red))
    plt.scatter(x, y, c='g')
    #result           
    x, y = list(zip(*yellow))
    plt.scatter(x, y, c='y')
